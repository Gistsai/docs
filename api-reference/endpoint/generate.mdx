---
title: "Generate"
api: "POST /generate/{gistId}/{variantSlug}"
description: "This endpoint calls the gist and the variant specified in the url parameter"
---

### Body

<ParamField body="values" type="json">
  A dictionary of variable names and values for the variables in the gist.
</ParamField>

### Response

<ResponseField name="error" type="string">
  Indicates whether the call failed with error
</ResponseField>

<ResponseField name="data" type="object">
  The contents of the user group
  <Expandable title="data" defaultOpen>
  <ResponseField name="id" type="string">
   The gist call id. Used for troubleshooting and tracing errors.
  </ResponseField>
    <ResponseField name="content" type="string">
      The gist output
    </ResponseField>
  </Expandable>
</ResponseField>
